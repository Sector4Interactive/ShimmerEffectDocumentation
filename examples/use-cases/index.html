    Implementation Tips

    Dynamic Switching

    Switch effects based on game state:

    public void UpdateItemRarity(ItemRarity rarity)
{
    switch(rarity)
    {
        case ItemRarity.Common:
            SetShimmer(Color.white, 2.0f, "Diagonal1", 0.2f);
            break;
        case ItemRarity.Rare:
            SetShimmer(Color.blue, 1.8f, "Diagonal2", 0.4f);
            break;
        case ItemRarity.Epic:
            SetShimmer(new Color(0.5f, 0f, 0.5f), 1.5f, "Diagonal3", 0.6f);
            break;
        case ItemRarity.Legendary:
            SetShimmer(Color.yellow, 1.2f, "Pulse", 0.8f);
            break;
    }
}

    Contextual Effects

    Adapt shimmer to game context:

    void UpdateShimmerBasedOnPlayerProximity()
{
    float distance = Vector3.Distance(player.position, transform.position);

    if (distance < 5f)
    {
        // Close: bright and fast
        material.SetFloat("_Speed", 3.0f);
        material.SetFloat("_Emission", 0.8f);
    }
    else if (distance < 15f)
    {
        // Medium: standard
        material.SetFloat("_Speed", 2.0f);
        material.SetFloat("_Emission", 0.4f);
    }
    else
    {
        // Far: subtle
        material.SetFloat("_Speed", 1.0f);
        material.SetFloat("_Emission", 0.2f);
    }
}
